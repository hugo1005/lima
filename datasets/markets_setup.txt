CREATE TABLE IF NOT EXISTS prices (
    timestamp real,
    exchange text,
    ticker text,
    best_bid real,
    best_ask real, 
    l2_ask real,
    l3_ask real,
    l4_ask real,
    l5_ask real, 
    l2_bid real,
    l3_bid real,
    l4_bid real,
    l5_bid real,
    l2_ask_vol real,
    l3_ask_vol real,
    l4_ask_vol real,
    l5_ask_vol real, 
    l2_bid_vol real,
    l3_bid_vol real,
    l4_bid_vol real,
    l5_bid_vol real, 
    bid_depth integer, 
    ask_depth integer, 
    bid_volume real, 
    ask_volume real, 
    n_bids integer, 
    n_asks integer,
    PRIMARY KEY (timestamp, exchange, ticker)
);
